getPDOType(string $type) : integer
$driver = new Driver();

$type = $driver->getPDOType(getType(2));
结果：PDO::PARAM_INT (1)

$type = $driver->getPDOType(getType('a'));
结果：PDO::PARAM_STR (2)

$type = $driver->getPDOType(getType(true));
结果：PDO::PARAM_BOOL (5)

$type = $driver->getPDOType(getType(NULL));
结果：PDO::PARAM_NULL (0)
